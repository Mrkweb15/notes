Imports MySql.Data.MySqlClient
Public Class Form1


Dim myConn As New MySqlConnection(&quot;Server=localhost; User ID=root; Password=; Database=sample&quot;)
Dim myDT As New DataTable


Private Sub Form1_Load(sender As Object, e As EventArgs) Handles MyBase.Load
	display()
End Sub


Public Sub display()
	Dim myAdapter As New MySqlDataAdapter(&quot;Select * from students&quot;, myConn)

	myDT.Clear()
	myAdapter.Fill(myDT)

	myGrid.DataSource = myDT
End Sub


Public Sub Clear()
	txtID.Clear()
	txtLname.Clear()
	txtFname.Clear()
	txtCourse.Clear()
	txtEmail.Clear()
End Sub


Private Sub btnView_Click(sender As Object, e As EventArgs) Handles btnView.Click
	display()
End Sub


Private Sub txtSearch_TextChanged(sender As Object, e As EventArgs) Handles txtSearch.TextChanged
	Dim searchAdapter As New MySqlDataAdapter(&quot;Select * from students where lname like &#39;%&quot; &amp; txtSearch.Text &amp; &quot;%&#39; or fname like &#39;%&quot;
	&amp; txtSearch.Text &amp; &quot;%&#39;&quot;, myConn)
	myDT.Clear()

	searchAdapter.Fill(myDT)
	myGrid.DataSource = myDT
End Sub


Private Sub btnAdd_Click(sender As Object, e As EventArgs) Handles btnAdd.Click
	Dim strSQL As String

	strSQL = &quot;Insert Into students values(&#39;&quot; &amp; txtID.Text &amp; &quot;&#39;, &#39;&quot; &amp; txtLname.Text &amp; &quot;&#39;, &#39;&quot; &amp; txtFname.Text &amp; &quot;&#39;, &#39;&quot; &amp;
	txtCourse.Text &amp; &quot;&#39;, &#39;&quot; &amp; txtEmail.Text &amp; &quot;&#39;)&quot;

	Dim myCMD As New MySqlCommand
	Dim checkAdapter As New MySqlDataAdapter(&quot;select * from students where idnum = &#39;&quot; &amp; txtID.Text &amp; &quot;&#39;&quot;, myConn)

	myDT.Clear()

	checkAdapter.Fill(myDT)

	If myDT.Rows.Count &gt; 0 Then
	MsgBox(&quot;Record already exists!&quot;, MsgBoxStyle.Critical, &quot;Duplicate Record&quot;)
	Exit Sub
	End If

	If myConn.State = ConnectionState.Closed Then
	myConn.Open()
	End If

	myCMD.Connection = myConn
	myCMD.CommandText = strSQL

	myCMD.ExecuteNonQuery()
	myConn.Close()
	display()
	Clear()

End Sub


Private Sub btnEdit_Click(sender As Object, e As EventArgs) Handles btnEdit.Click
	Dim selRow As Integer
	Dim strSQL As String
	Dim myCMd As New MySqlCommand

	selRow = myGrid.CurrentRow.Index

	If btnEdit.Text = &quot;Edit&quot; Then
	txtID.Text = myGrid.Item(0, selRow).Value
	txtLname.Text = myGrid.Item(1, selRow).Value
	txtFname.Text = myGrid.Item(2, selRow).Value
	txtCourse.Text = myGrid.Item(3, selRow).Value
	txtEmail.Text = myGrid.Item(4, selRow).Value
	btnEdit.Text = &quot;Save&quot;

	Else
	&#39;code for updating record
	strSQL = &quot;Update students set lname = &#39;&quot; &amp; txtLname.Text &amp; &quot;&#39;, fname = &#39;&quot; &amp; txtFname.Text &amp; &quot;&#39;, course = &#39;&quot; &amp; txtCourse.Text
	&amp; &quot;&#39;, email = &#39;&quot; &amp; txtEmail.Text &amp; &quot;&#39; where idnum = &#39;&quot; &amp; txtID.Text &amp; &quot;&#39;&quot;
		If myConn.State = ConnectionState.Closed Then
		myConn.Open()

		End If

	myCMd.Connection = myConn
	myCMd.CommandText = strSQL

	myCMd.ExecuteNonQuery()
	myConn.Close()
	MsgBox(&quot;Record Updated&quot;)
	display()
	Clear()
	btnEdit.Text = &quot;Edit&quot;

	End If

End Sub


Private Sub btnDelete_Click(sender As Object, e As EventArgs) Handles btnDelete.Click
	Dim strSQL As String
	Dim selRow As Integer
	Dim mycmd As New MySqlCommand

	Dim ans As Integer
	selRow = myGrid.CurrentRow.Index
	ans = MsgBox(&quot;Are you sure you want to delete Record of Student: &quot; + myGrid.Item(0, selRow).Value, MsgBoxStyle.YesNo +
	MsgBoxStyle.Question, &quot;Delete Record&quot;)

	If ans = 6 Then
	strSQL = &quot;Delete from students where idnum = &#39;&quot; &amp; myGrid.Item(0, selRow).Value &amp; &quot;&#39;&quot;
		If myConn.State = ConnectionState.Closed Then
		myConn.Open()
		End If
	mycmd.Connection = myConn
	mycmd.CommandText = strSQL
	mycmd.ExecuteNonQuery()
	myConn.Close()
	display()
	End If

End Sub
End Class
